```
git clone https://github.com/mrlihx/epay.git ./
```

```
git clone https://ghproxy.com/https://github.com/mrlihx/epay.git ./
```

```
git pull
```

### Nginx 伪静态
```
location / {
 if (!-e $request_filename) {
   rewrite ^/(.[a-zA-Z0-9\-\_]+).html$ /index.php?mod=$1 last;
 }
 rewrite ^/pay/(.*)$ /pay.php?s=$1 last;
}
location ^~ /plugins {
  deny all;
}
location ^~ /includes {
  deny all;
}
```
